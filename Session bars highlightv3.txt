//@version=6
indicator('Session bars V3 最簡版', '', true) // 疊加主K線，名稱可自定
timeRange = input.string('0915-1630', title="交易時段") // 自定HHMM-HHMM，無需改程式
// 核心：判斷是否為【≥4H周期/日/周/月】→ 是則屏蔽背景色
isHide = (timeframe.period == 'H' and timeframe.multiplier >= 4) or timeframe.isdaily or timeframe.isweekly or timeframe.ismonthly
// 判斷當前K線是否在指定時段內
inSession = not na(time(timeframe.period, timeRange))
// 僅【1H及以下周期 + 時段內】繪製淺綠半透明背景，其他情況不繪製
bgcolor(not isHide and inSession ?na: color.new(color.green, 60))
